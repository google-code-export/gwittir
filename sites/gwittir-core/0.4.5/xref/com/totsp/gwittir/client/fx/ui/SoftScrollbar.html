40  public class SoftScrollbar extends Composite {
41      VerticalPanel vertical = new VerticalPanel();
42      FocusPanel barTarget = new FocusPanel();
43      FocusPanel higherTarget = new FocusPanel();
44      FocusPanel lowerTarget = new FocusPanel();
45      ScrollListener scrollListener = new ScrollListener() {
46              public void onScroll(Widget widget, int scrollLeft, int scrollTop) {
47                  refresh();
48              }
49          };
50  
51      SimplePanel bar = new SimplePanel();
52      SimplePanel base;
53      SimplePanel higher = new SimplePanel();
54      SimplePanel lower = new SimplePanel();
55      SoftScrollArea target;
56      private MouseListener barListener = new MouseListenerAdapter() {
57              boolean inDrag = false;
58              int start;
59  
60              public void onMouseLeave(Widget sender) {
61                  inDrag = false;
62              }
63  
64              public void onMouseUp(Widget sender, int x, int y) {
65                  inDrag = false;
66              }
67  
68              public void onMouseDown(Widget sender, int x, int y) {
69                  inDrag = true;
70                  start = y;
71              }
72  
73              public void onMouseMove(Widget sender, int x, int y) {
74                  if(inDrag) {
75                      float newPercent = (float) (
76                              (y - start) + lower.getOffsetHeight()
77                          ) / (float) base.getOffsetHeight();
78                      int newPosition = Math.round((
79                                  target.getOffsetHeight()
80                                  + target.getMaxScrollPosition()
81                              ) * newPercent) + x;
82                      target.setScrollPosition(newPosition);
83                  }
84              }
85          };
86  
87      private MouseListener higherListener = new MouseListenerAdapter() {
88              public void onMouseDown(Widget sender, int x, int y) {
89                  float newPercent = (float) (
90                          y + lower.getOffsetHeight() + bar.getOffsetHeight()
91                      ) / (float) base.getOffsetHeight();
92                  int newPosition = Math.round((
93                              target.getOffsetHeight()
94                              + target.getMaxScrollPosition()
95                          ) * newPercent) - (target.getOffsetHeight() / 2);
96  
97                  target.setScrollPosition(newPosition);
98              }
99          };
100 
101     private MouseListener lowerListener = new MouseListenerAdapter() {
102             public void onMouseDown(Widget sender, int x, int y) {
103                 float newPercent = (float) y / (float) base.getOffsetHeight();
104                 int newPosition = Math.round((
105                             target.getOffsetHeight()
106                             + target.getMaxScrollPosition()
107                         ) * newPercent) - (target.getOffsetHeight() / 2);
108                 target.setScrollPosition(newPosition);
109             }
110         };
111 
112     private WindowResizeListener windowListener = new WindowResizeListener() {
113             public void onWindowResized(int width, int height) {
114                 refresh();
115             }
116         };

118     /** Creates a new instance of SoftVerticalScroll */
119     public SoftScrollbar(SoftScrollArea target) {
120         this.target = target;
121         this.base = new SimplePanel();
122 
123         
124         this.base.setWidget(vertical);
125         super.initWidget(this.base);
126         this.higherTarget.setWidget(this.higher);
127         this.barTarget.setWidget(this.bar);
128         this.lowerTarget.setWidget(this.lower);
129 
130         vertical.add(lowerTarget);
131         vertical.add(barTarget);
132         vertical.add(higherTarget);
133         this.setStyleName("gwittir-SoftScrollbar");
134         this.bar.setStyleName("bar");
135         this.lower.setStyleName("lower");
136         this.higher.setStyleName("higher");
137         DOM.setStyleAttribute(this.lower.getElement(), "overflow", "hidden");
138         DOM.setStyleAttribute(this.bar.getElement(), "overflow", "hidden");
139         DOM.setStyleAttribute(this.higher.getElement(), "overflow", "hidden");
140         DOM.setStyleAttribute(this.lowerTarget.getElement(), "overflow", "hidden");
141         DOM.setStyleAttribute(this.barTarget.getElement(), "overflow", "hidden");
142         DOM.setStyleAttribute(this.higherTarget.getElement(), "overflow", "hidden");
143         DOM.setStyleAttribute(this.getElement(), "overflow", "hidden");
144     }

158     protected void onAttach() {
159         super.onAttach();
160         target.addScrollListener(this.scrollListener);
161         this.lowerTarget.addMouseListener(this.getLowerListener());
162         this.higherTarget.addMouseListener(this.getHigherListener());
163         this.barTarget.addMouseListener(this.getBarListener());
164         Window.addWindowResizeListener(this.windowListener);
165 
166         Timer t = new Timer() {
167                 public void run() {
168                     refresh();
169                 }
170             };
171 
172         t.schedule(10);
173     }
174 
175     protected void onDetach() {
176         super.onDetach();
177         Window.removeWindowResizeListener(this.windowListener);
178         target.removeScrollListener(this.scrollListener);
179         this.lowerTarget.removeMouseListener(this.getLowerListener());
180         this.higherTarget.removeMouseListener(this.getHigherListener());
181         this.barTarget.removeMouseListener(this.getBarListener());
182     }

184     public void refresh() {
185         int currentHeight = this.getOffsetHeight();
186 
187         float pageSize = (float) target.getOffsetHeight() / (float) (
188                 target.getMaxScrollPosition() + target.getOffsetHeight()
189             );
190 
191         //GWT.log("Page sisze " + pageSize, null);
192         float scrollPercentage = (float) target.getScrollPosition() / (
193                 target.getMaxScrollPosition() + 1
194             );
195 
196         int lowerHeight = Math.round((
197                     currentHeight
198                     - Math.round((float) currentHeight * pageSize)
199                 ) * scrollPercentage);
200         lowerHeight = lowerHeight == 0 ? 1 : lowerHeight; // Fixes MSIE;
201         this.lower.setHeight(lowerHeight + "px");
202         int barHeight = Math.round(currentHeight * pageSize);
203         this.bar.setHeight(barHeight + "px");
204         int higherHeight = currentHeight - lowerHeight - barHeight;
205         higherHeight = higherHeight < 0 ? 0 : higherHeight;
206         this.higher.setHeight(higherHeight + "px");
207         if( this.getOffsetWidth() >= 0 ){
208             this.lower.setWidth(this.getOffsetWidth() + "px");
209             this.bar.setWidth(this.getOffsetWidth() + "px");
210             this.higher.setWidth(this.getOffsetWidth() + "px");
211         }
212     }
